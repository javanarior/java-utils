16  package de.javanarior.utils.compiler;
17  
18  import java.util.Hashtable;
19  import java.util.Map;
20  
21  /**
22   * Class loader to load {@link MemoryJavaByteCode}. The ClassLoader follows the 'parent-first' delegation model .
23   */
24  class MemoryClassLoader extends ClassLoader {
25  
26      private static final Map<String, Class<?>> CLASSES = new Hashtable<>();
27  
28      /**
29       * Create a new MemoryClassLoader instance.
30       *
31       * @param parent
32       *            - parent classloader
33       */
34      public MemoryClassLoader(ClassLoader parent) {
35          super(parent);
36      }
37  
38      /**
39       * @{inherit_doc
40       */
41      @Override
42      protected Class<?> findClass(String name) throws ClassNotFoundException {
43          Class<?> clazz = CLASSES.get(name);
44          if (clazz != null) {
45              return clazz;
46          }
47          return super.findClass(name);
48      }
49  
50      /**
51       * Load/find class specified by {@link MemoryJavaByteCode}.
52       *
53       * @param byteCode
54       *            - {@code MemoryJavaByteCode} to load class from
55       * @return {@code Class} object
56       */
57      public Class<?> load(MemoryJavaByteCode byteCode) {
58          String binaryName = byteCode.getBinaryName();
59          Class<?> loadedClass = findLoadedClass(binaryName);
60          if (loadedClass != null) {
61              return loadedClass;
62          }
63          try {
64              return loadClass(binaryName);
65          } catch (ClassNotFoundException exception) {
66              Class<?> defineClass = defineClass(binaryName, byteCode.getBytes(), 0, byteCode.getBytes().length);
67              resolveClass(defineClass);
68              CLASSES.put(binaryName, defineClass);
69              return defineClass;
70          }
71      }
72  
73  }
