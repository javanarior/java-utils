16  package de.javanarior.utils.compiler;
17  
18  import java.io.IOException;
19  
20  import javax.tools.FileObject;
21  import javax.tools.ForwardingJavaFileManager;
22  import javax.tools.JavaFileManager;
23  import javax.tools.JavaFileObject;
24  import javax.tools.JavaFileObject.Kind;
25  import javax.tools.StandardJavaFileManager;
26  
27  /**
28   * More as simple File Manager for in Memory compilation.
29   */
30  class MemoryFileManager extends ForwardingJavaFileManager<JavaFileManager> {
31  
32      private MemoryJavaByteCode classObject;
33  
34      /**
35       * Create a new MemoryFileManager instance.
36       *
37       * @param standardJavaFileManager
38       *            - Standard File Manager
39       */
40      public MemoryFileManager(StandardJavaFileManager standardJavaFileManager) {
41          super(standardJavaFileManager);
42      }
43  
44      /**
45       * @{inherit_doc
46       */
47      @Override
48      public JavaFileObject getJavaFileForOutput(Location location, String className, Kind kind, FileObject sibling)
49                      throws IOException {
50          if (classObject == null) {
51              classObject = new MemoryJavaByteCode(className, kind);
52          }
53          return classObject;
54      }
55  
56      /**
57       * @{inherit_doc
58       */
59      @Override
60      public FileObject getFileForOutput(Location location, String packageName, String relativeName, FileObject sibling)
61                      throws IOException {
62          return getJavaFileForOutput(location, packageName + "." + relativeName, Kind.CLASS, sibling);
63      }
64  
65  }
